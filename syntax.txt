<identifier> ::= 'a...Z' | '_' # Any character from a to Z, and _.
<shift_operator> ::= '<-'
<end_of_statement> ::= '->'
<keyword> ::= 'i' | 'o'
<if> ::= '?'
<else_if> ::= '??'
<else> ::= '???'

<statement> ::= <expression-statement>
              | <compound-statement>
              | <selection-statement>
              | <iteration-statement>

<expression_statement> ::= 'python_expression' <end_of_statement> | <expression> <end_of_statement>
<expression> ::= <assignment_expression> | <function_call>
<assignment_expression> ::= <identifier> <shift_operator> 

<function_call> ::= '[' <identifier> ']' <shift_operator> <argument_list> | 
	'[' <keyword> ']' <shift_operator> <argument_list>

<argument_list> ::= 'python_expression' | 'python_expression' <shift_operator> <argument_list>

<compound_statement> ::= <if_statement> | <function_definition>
